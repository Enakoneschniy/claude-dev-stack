```javascript
async function loadTimeline() {
  const res = await fetch('/api/sessions');
  const sessions = await res.json();
  const list = document.getElementById('timeline-list');
  // Clear existing children safely
  while (list.firstChild) list.removeChild(list.firstChild);
  sessions.forEach(s => {
    const card = document.createElement('div');
    card.className = 'session-card';
    const date = document.createElement('div');
    date.className = 'session-date';
    date.textContent = new Date(s.start_time).toLocaleString();
    const summary = document.createElement('div');
    summary.className = 'session-summary';
    summary.textContent = s.summary || 'No summary';
    const project = document.createElement('div');
    project.className = 'session-project';
    project.textContent = s.project;
    card.appendChild(date);
    card.appendChild(project);
    card.appendChild(summary);
    list.appendChild(card);
  });
}
```

```javascript
async function loadCosts() {
  const res = await fetch('/api/costs');
  const data = await res.json();
  
  document.getElementById('stat-cost').textContent = '$' + data.totals.costUsd.toFixed(2);
  
  const recent = data.sessions.slice(0, 20);

  new Chart(document.getElementById('chart-tokens'), {

    data: {
      labels,
      datasets: [
        { label: 'Input Tokens', data: recent.map(s => s.inputTokens), backgroundColor: 'rgba(41, 151, 255, 0.7)' },
        { label: 'Output Tokens', data: recent.map(s => s.outputTokens), backgroundColor: 'rgba(52, 199, 89, 0.7)' }
      ]
    },
    options: {
      responsive: true,
      scales: { x: { stacked: true }, y: { stacked: true, beginAtZero: true } }
```

**5. Entity graph with Cytoscape.js:**
```javascript
let graphLoaded = false;
let cy = null;

async function loadGraph() {
  const res = await fetch('/api/graph');
  const data = await res.json();
  
  const elements = [
    ...data.nodes.map(n => ({
      data: { id: String(n.id), label: n.displayName || n.name, type: n.type }
    })),
    ...data.edges.map((e, i) => ({
      data: { id: 'e' + i, source: String(e.from), target: String(e.to), label: e.relationType, weight: e.weight }
    }))
  ];
  
  const typeColors = {
    person: '#ff3b30', project: '#0071e3', concept: '#34c759',
    decision: '#ff9500', file: '#5ac8fa', commit: '#af52de',
    skill: '#ffcc00', api: '#5856d6'
  };
  
  cy = cytoscape({
    container: document.getElementById('graph-container'),
    elements,
    style: [
      {
        selector: 'node',
        style: {
          'label': 'data(label)',
          'background-color': function(ele) { return typeColors[ele.data('type')] || '#6e6e73'; },
          'color': '#1d1d1f',
          'font-size': '10px',
          'text-valign': 'bottom',
          'text-margin-y': 4,
          'width': 30,
          'height': 30
        }
      },
      {
        selector: 'edge',
        style: {
          'width': function(ele) { return Math.min(ele.data('weight') || 1, 5); },
          'line-color': '#d2d2d7',
          'target-arrow-color': '#d2d2d7',
          'target-arrow-shape': 'triangle',
          'curve-style': 'bezier'
        }
      },
      {
        selector: ':selected',
        style: { 'border-width': 2, 'border-color': '#0071e3' }
      }
    ],
    layout: { name: 'cose', idealEdgeLength: 100, nodeOverlap: 20, animate: true, animationDuration: 500 }
  });
  
  // Click handler — uses textContent only (T-48-06: XSS safe)
  cy.on('tap', 'node', function(evt) {
    const node = evt.target;
    const details = document.getElementById('node-details');
    details.classList.remove('hidden');
    document.getElementById('node-name').textContent = node.data('label');
    document.getElementById('node-type').textContent = 'Type: ' + node.data('type');
    const neighbors = node.neighborhood('node');
    document.getElementById('node-connections').textContent = 
      'Connections: ' + neighbors.length + ' (' + neighbors.map(n => n.data('label')).join(', ') + ')';
  });
  
  cy.on('tap', function(evt) {
    if (evt.target === cy) {
      document.getElementById('node-details').classList.add('hidden');
    }
  });
  
  graphLoaded = true;
}
```

```javascript
#!/usr/bin/env node
// Post-build script: copies dashboard-assets/ from src/ to dist/
import { cpSync, existsSync, mkdirSync } from 'node:fs';
import { join, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';

const __dirname = dirname(fileURLToPath(import.meta.url));
const src = join(__dirname, 'dashboard-assets');
const dest = join(__dirname, '..', 'dist', 'dashboard-assets');

if (!existsSync(src)) {
  console.error('Dashboard assets source not found:', src);
  process.exit(1);
}

mkdirSync(dest, { recursive: true });
cpSync(src, dest, { recursive: true });
console.log('Dashboard assets copied to', dest);
```
