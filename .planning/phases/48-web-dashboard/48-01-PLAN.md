## Threat Model (ASVS L1)

| ID | Threat | Severity | Mitigation |
|----|--------|----------|------------|
| T-48-01 | Path traversal via serveStatic | HIGH | serveStatic root is an absolute resolved path to dist/dashboard-assets/; no user input in path construction |
| T-48-02 | Session data exposure over network | MEDIUM | Server binds to 127.0.0.1 (localhost only), not 0.0.0.0; no remote access possible |
| T-48-03 | PID file race condition | LOW | Check process liveness with process.kill(pid, 0) before trusting PID file; use try/catch for ESRCH |
| T-48-04 | SQL injection via API params | MEDIUM | All DB queries use prepared statements (already enforced in sessions.ts); route params validated as integers |
| T-48-05 | DoS via large graph response | LOW | Limit graph nodes to top 500 by relation count; response size bounded |
</threat_model>

**4. API route factory:**
```typescript
function createApp(projectPath: string): Hono {
  const app = new Hono();

  // Static assets — resolve to dist/dashboard-assets/ relative to this file
  const assetsDir = resolve(dirname(new URL(import.meta.url).pathname), 'dashboard-assets');
  app.use('/*', serveStatic({ root: assetsDir }));

  // API routes
  app.get('/api/sessions', (c) => {
    const db = openSessionsDB(projectPath);
    try {
      const sessions = db.listSessions({ limit: 100 });
      return c.json(sessions);
    } finally {
      closeSessionsDB(projectPath);
    }
  });

  app.get('/api/sessions/:id/observations', (c) => {
    const sessionId = c.req.param('id');
    const db = openSessionsDB(projectPath);
    try {
      const observations = db.listObservations({ sessionId, limit: 200 });
      return c.json(observations);
    } finally {
      closeSessionsDB(projectPath);
    }
  });

  app.get('/api/stats', (c) => {
    const db = openSessionsDB(projectPath);
    try {
      const sessions = db.listSessions({ limit: 1000 });
      const countsByType = db.countObservationsByType();
      const entityCount = db.countEntities();
      const topEntities = db.topEntities(10);
      const totalObs = countsByType.reduce((s, r) => s + r.count, 0);
      return c.json({
        sessionCount: sessions.length,
        observationCount: totalObs,
        observationsByType: countsByType,
        entityCount,
        topEntities,
      });
    } finally {
      closeSessionsDB(projectPath);
    }
  });

  app.get('/api/costs', (c) => {
    const db = openSessionsDB(projectPath);
    try {
      const sessions = db.listSessions({ limit: 100 });
      const costs = sessions.map(s => {
        const obs = db.listObservations({ sessionId: s.id, limit: 500 });
        let inputTokens = 0;
        let outputTokens = 0;
        let costUsd = 0;
        for (const o of obs) {
          // Parse cost data from observation content (captured by Stop hook)
          const inputMatch = o.content.match(/input[_ ]tokens?:?\s*([\d,]+)/i);
          const outputMatch = o.content.match(/output[_ ]tokens?:?\s*([\d,]+)/i);
          const costMatch = o.content.match(/cost(?:_usd)?:?\s*\$?([\d.]+)/i);
          if (inputMatch) inputTokens += parseInt(inputMatch[1].replace(/,/g, ''), 10);
          if (outputMatch) outputTokens += parseInt(outputMatch[1].replace(/,/g, ''), 10);
          if (costMatch) costUsd += parseFloat(costMatch[1]);
        }
        return {
          sessionId: s.id,
          date: s.start_time,
          project: s.project,
          summary: s.summary,
          inputTokens,
          outputTokens,
          costUsd,
        };
      });
      const totalCost = costs.reduce((s, c) => s + c.costUsd, 0);
      const totalInput = costs.reduce((s, c) => s + c.inputTokens, 0);
      const totalOutput = costs.reduce((s, c) => s + c.outputTokens, 0);
      return c.json({ sessions: costs, totals: { costUsd: totalCost, inputTokens: totalInput, outputTokens: totalOutput } });
    } finally {
      closeSessionsDB(projectPath);
    }
  });

  app.get('/api/graph', (c) => {
    const graph = getEntityGraph(projectPath);
    // Limit to top 500 nodes by edge count for performance (T-48-05)
    if (graph.nodes.length > 500) {
      const edgeCounts = new Map<number, number>();
      for (const e of graph.edges) {
        edgeCounts.set(e.from, (edgeCounts.get(e.from) ?? 0) + 1);
        edgeCounts.set(e.to, (edgeCounts.get(e.to) ?? 0) + 1);
      }
      const topIds = new Set(
        [...edgeCounts.entries()]
          .sort((a, b) => b[1] - a[1])
          .slice(0, 500)
          .map(([id]) => id)
      );
      graph.nodes = graph.nodes.filter(n => topIds.has(n.id));
      graph.edges = graph.edges.filter(e => topIds.has(e.from) && topIds.has(e.to));
    }
    return c.json(graph);
  });

  return app;
}
```

**5. Exported functions:**

`startDashboard(options: { port?: number; projectPath?: string; noBrowser?: boolean }): Promise<void>`:
- Check for stale PID file, clean if needed
- If PID file exists and process running, print "Dashboard already running at http://localhost:{port}" and return
- Create Hono app with `createApp(projectPath)`
- Start server with `serve({ fetch: app.fetch, port, hostname: '127.0.0.1' })` (T-48-02: localhost only)
- Write PID file with `process.pid`
- Register SIGINT and SIGTERM handlers that call `shutdown()`
- If `!noBrowser`, dynamic import `open` and open `http://localhost:{port}`
- Print "Dashboard running at http://localhost:{port}"

`stopDashboard(): Promise<void>`:
- Read PID file; if null, print "No dashboard running" and return
- If process running, `process.kill(pid, 'SIGTERM')`
- Wait up to 3 seconds for process to exit (poll with `isProcessRunning`)
- Delete PID file
- Print "Dashboard stopped"

`shutdown(server: ReturnType<typeof serve>): void`:
- Close server
- Delete PID file
- `process.exit(0)`

**6. CLI entry point:**

```typescript
export async function main(args: string[]): Promise<void> {
  if (args[0] === 'stop') {
    await stopDashboard();
    return;
  }
  const portFlag = args.indexOf('--port');
  const port = portFlag !== -1 ? parseInt(args[portFlag + 1], 10) : DEFAULT_PORT;
  const noBrowser = args.includes('--no-browser');
  await startDashboard({ port, projectPath: process.cwd(), noBrowser });
}
```
