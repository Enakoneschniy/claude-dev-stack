# Plan 01: Credential Resolver with Three-Path Fallback

<objective>
Create a credential resolver module in @cds/core that tries OAuth token, stored API key, and ANTHROPIC_API_KEY env var in order. Integrate it into agent-dispatcher.ts replacing the current two-step inline check. Add CredentialError to the error hierarchy.
</objective>

<must_haves>
- Credential resolver tries three sources in order: OAuth → stored key → env var
- Clear error message when all three fail, listing attempted sources
- agent-dispatcher.ts uses resolveCredential() instead of inline ANTHROPIC_API_KEY check
- OAuth path works on Linux (filesystem-based, no macOS Keychain dependency)
- CredentialError exported from @cds/core
</must_haves>

<threat_model>
## Threat Model

| ID | Threat | Severity | Mitigation |
|----|--------|----------|------------|
| T-46-01 | Credential leakage in logs | high | Never log the resolved API key value; log only the source type ('oauth', 'api-key-stored', 'api-key-env') |
| T-46-02 | Credential file readable by other users | medium | Check file permissions (0600) on stored credential files; warn if world-readable |
| T-46-03 | Temporary env var mutation visible to child processes | medium | Use process.env assignment scoped to the dispatch call; restore original value after SDK query completes |
| T-46-04 | OAuth token expiry not detected | low | If OAuth token exists but SDK returns auth error, fall through to next source with warning |
</threat_model>

```typescript
/**
 * credential-resolver.ts — Three-path credential fallback chain.
 *
 * Phase 46 (HARD-05): resolves API credentials from multiple sources:
 *   1. OAuth token from Claude Code session (~/.claude/ filesystem)
 *   2. Stored API key from ~/.claude-dev-stack/credentials.json
 *   3. ANTHROPIC_API_KEY environment variable
 *
 * Design:
 *   - Tries sources in order; first success wins
 *   - Never logs credential values — only source type
 *   - Linux-compatible: filesystem only (no macOS Keychain)
 *   - Throws CredentialError listing all attempted sources on total failure
 */

import { readFileSync, statSync } from 'node:fs';
import { join } from 'node:path';
import { homedir } from 'node:os';
import { execFileSync } from 'node:child_process';
import { CredentialError } from './errors.js';

export interface ResolvedCredential {
  type: 'oauth' | 'api-key-stored' | 'api-key-env';
  apiKey: string;
  source: string;
}

/**
 * Attempt to extract an API key from Claude Code's OAuth session.
 * Checks ~/.claude/ for credential files (Linux-compatible, no Keychain).
 * Falls back to `claude auth print-api-key` subprocess if available.
 */
function tryOAuth(): ResolvedCredential | null {
  // Strategy 1: Check Claude Code config files for stored auth
  const claudeDir = join(homedir(), '.claude');
  const credPaths = [
    join(claudeDir, 'credentials.json'),
    join(claudeDir, 'auth.json'),
  ];

  for (const credPath of credPaths) {
    try {
      const raw = readFileSync(credPath, 'utf8');
      const data = JSON.parse(raw) as Record<string, unknown>;
      const key =
        (data.api_key as string) ??
        (data.apiKey as string) ??
        (data.anthropic_api_key as string);
      if (typeof key === 'string' && key.length > 0) {
        return { type: 'oauth', apiKey: key, source: `OAuth file: ${credPath}` };
      }
    } catch {
      // File doesn't exist or isn't valid JSON — try next
    }
  }

  // Strategy 2: Use Claude CLI to extract auth (works if Claude Code is installed)
  try {
    const output = execFileSync('claude', ['auth', 'print-api-key'], {
      encoding: 'utf8',
      stdio: 'pipe',
      timeout: 5000,
    }).trim();
    if (output.length > 0 && output.startsWith('sk-')) {
      return { type: 'oauth', apiKey: output, source: 'OAuth via `claude auth print-api-key`' };
    }
  } catch {
    // Claude CLI not available or auth command failed
  }

  return null;
}

/**
 * Check for a stored API key in ~/.claude-dev-stack/credentials.json.
 * Warns (via console.warn) if the file has world-readable permissions.
 */
function tryStoredKey(): ResolvedCredential | null {
  const credPath = join(homedir(), '.claude-dev-stack', 'credentials.json');

  try {
    // Permission check (T-46-02 mitigation)
    const stat = statSync(credPath);
    const mode = stat.mode & 0o777;
    if (mode & 0o044) {
      console.warn(
        `Warning: ${credPath} is readable by other users (mode ${mode.toString(8)}). ` +
        `Run: chmod 600 ${credPath}`,
      );
    }

    const raw = readFileSync(credPath, 'utf8');
    const data = JSON.parse(raw) as Record<string, unknown>;
    const key =
      (data.api_key as string) ??
      (data.apiKey as string) ??
      (data.ANTHROPIC_API_KEY as string);
    if (typeof key === 'string' && key.length > 0) {
      return { type: 'api-key-stored', apiKey: key, source: `Stored key: ${credPath}` };
    }
  } catch {
    // File doesn't exist or isn't valid JSON
  }

  return null;
}

/**
 * Check ANTHROPIC_API_KEY environment variable.
 */
function tryEnvVar(): ResolvedCredential | null {
  const key = process.env.ANTHROPIC_API_KEY;
  if (typeof key === 'string' && key.length > 0) {
    return { type: 'api-key-env', apiKey: key, source: 'Environment variable: ANTHROPIC_API_KEY' };
  }
  return null;
}

/**
 * Resolve API credentials from three sources in priority order:
 * 1. OAuth token (Claude Code session / stored OAuth credentials)
 * 2. Stored API key (~/.claude-dev-stack/credentials.json)
 * 3. ANTHROPIC_API_KEY environment variable
 *
 * @throws CredentialError when all three sources fail
 */
export function resolveCredential(): ResolvedCredential {
  const sources: string[] = [];

  // 1. OAuth
  const oauth = tryOAuth();
  if (oauth) return oauth;
  sources.push('OAuth (Claude Code session files + CLI)');

  // 2. Stored key
  const stored = tryStoredKey();
  if (stored) return stored;
  sources.push('Stored key (~/.claude-dev-stack/credentials.json)');

  // 3. Environment variable
  const env = tryEnvVar();
  if (env) return env;
  sources.push('Environment variable (ANTHROPIC_API_KEY)');

  throw new CredentialError(sources);
}
```

This module:
- Tries OAuth first (filesystem probe + CLI subprocess), then stored key file, then env var
- Checks file permissions on credential files (T-46-02)
- Never logs actual key values (T-46-01)
- Works on Linux (no macOS Keychain dependency)
- Throws CredentialError with all attempted sources listed
