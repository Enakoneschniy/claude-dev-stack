    ```javascript
    #!/usr/bin/env node
    /**
     * hooks/idea-capture-trigger.mjs — UserPromptSubmit hook (CAPTURE-02/03/04)
     *
     * Detects idea-trigger phrases (Russian + English) in user prompts and
     * emits a hint nudging Claude to invoke /gsd-note for capture.
     *
     * Per D-18..D-22 in vault cds-core-independence-plan.md.
     *
     * Design:
     *   - Fail-silent (exit 0) on empty stdin / malformed JSON / missing config
     *   - No npm dependencies — node stdlib only
     *   - ReDoS-safe: MAX_PROMPT_LEN=4096, escaped-literal alternation, no nested quantifiers
     *   - First-match-wins (Russian checked before English — deterministic order)
     *   - Cyrillic uses explicit boundary class (JS \b is ASCII-only)
     *   - Optional telemetry bump in ~/.claude/cds-stats.json (CAPTURE-04, opt-out by deleting file)
     */

    import { existsSync, readFileSync, writeFileSync, mkdirSync } from 'node:fs';
    import { join, dirname } from 'node:path';
    import { homedir } from 'node:os';
    import { fileURLToPath } from 'node:url';

    const MAX_PROMPT_LEN = 4096;
    const __dirname = dirname(fileURLToPath(import.meta.url));
    const TRIGGERS_PATH = join(__dirname, 'idea-capture-triggers.json');
    const STATS_DIR = join(homedir(), '.claude');
    const STATS_PATH = join(STATS_DIR, 'cds-stats.json');

    const CYR_BOUNDARY_HEAD = "(?:^|[\\s.,!?;:()\"'«»—-])";
    const CYR_BOUNDARY_TAIL = "(?:$|[\\s.,!?;:()\"'«»—-])";

    function escapeRegex(s) { return s.replace(/[.*+?^${}()|[\]\\]/g, '\\$&'); }

    function loadTriggers() { /* readFileSync + JSON.parse in try/catch, validate shape, return null on any issue */ }

    function bumpTelemetry() {
      try {
        mkdirSync(STATS_DIR, { recursive: true });
        let stats = {};
        if (existsSync(STATS_PATH)) {
          try { stats = JSON.parse(readFileSync(STATS_PATH, 'utf8')); } catch { stats = {}; }
        }
        if (typeof stats !== 'object' || stats === null) stats = {};
        stats.idea_capture_hints_fired = (Number(stats.idea_capture_hints_fired) || 0) + 1;
        writeFileSync(STATS_PATH, JSON.stringify(stats, null, 2) + '\n');
      } catch { /* fail-silent */ }
    }

    async function readStdin(timeoutMs = 500) { /* same pattern as dev-router.mjs */ }

    function findMatch(text, triggers) {
      // Russian first (explicit boundary class)
      for (const phrase of triggers.russian || []) {
        const re = new RegExp(`${CYR_BOUNDARY_HEAD}${escapeRegex(phrase)}${CYR_BOUNDARY_TAIL}`, 'i');
        if (re.test(text)) return phrase;
      }
      // English second (ASCII \b)
      for (const phrase of triggers.english || []) {
        const re = new RegExp(`\\b${escapeRegex(phrase)}\\b`, 'i');
        if (re.test(text)) return phrase;
      }
      return null;
    }

    async function main() {
      const raw = await readStdin();
      if (!raw) return;

      let payload;
      try { payload = JSON.parse(raw); } catch { return; }

      const prompt = typeof payload?.prompt === 'string' ? payload.prompt : '';
      if (!prompt) return;

      const triggers = loadTriggers();
      if (!triggers) return;

      const text = prompt.length > MAX_PROMPT_LEN ? prompt.slice(0, MAX_PROMPT_LEN) : prompt;

      const matched = findMatch(text, triggers);
      if (!matched) return;

      // Exact hint format per CAPTURE-03 / D-21
      const hint = `💡 IDEA-CAPTURE HINT: Detected trigger phrase "${matched}" in user message. Consider invoking /gsd-note to capture the idea to .planning/notes/.`;
      process.stdout.write(hint);

      bumpTelemetry();
    }

    main().catch(() => {}).finally(() => process.exit(0));
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| user → hook stdin | Untrusted user-submitted prompt string reaches the hook via Claude Code JSON payload. Must be treated as adversarial input (arbitrary length, arbitrary encoding, possible regex-pathological patterns). |
| hook → stdout | Trusted-by-Claude-Code: stdout content gets prepended to the model's context window. Hook author is CDS, Claude Code treats it as authoritative. Must not emit user-controlled strings verbatim. |
| hook → ~/.claude/cds-stats.json | Local filesystem write on user's machine. Low-value target (just a counter), but must handle EPERM/ENOSPC gracefully. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-32-01 | D (Denial of Service) | Regex matching in `findMatch()` | mitigate | MAX_PROMPT_LEN=4096 truncation before regex testing (copied from dev-router.mjs T-31-04 mitigation). Trigger list uses only literal-string alternations (no nested quantifiers, no backreferences) — ReDoS-immune by construction. |
| T-32-02 | T (Tampering) | User prompt echoed in hint string | mitigate | Hint uses the trigger phrase from the JSON config file (not the user's matched capture), so user can't inject arbitrary text into Claude's context via this hook. `{phrase}` placeholder is filled with the literal from `idea-capture-triggers.json`, which is committed code. |
| T-32-03 | I (Information Disclosure) | `~/.claude/cds-stats.json` file mode | accept | File contains only a hint-fired counter, no PII. No sensitive data. User already owns `~/.claude/` so default mode is fine. |
| T-32-04 | D (Denial of Service) | Hook failure blocks user message | mitigate | `main().catch(() => {}).finally(() => process.exit(0))` wrapper — hook ALWAYS exits 0 regardless of error. Claude Code continues normally on hook error. Telemetry block wrapped in try/catch; filesystem errors never propagate. |
| T-32-05 | E (Elevation of Privilege) | Arbitrary JSON config file | mitigate | `idea-capture-triggers.json` lives in `hooks/` directory (shipped with the package, not user-writable at install site unless user has modified their local checkout). Hook's `loadTriggers()` validates shape (`russian` + `english` must be string arrays); rejects anything else silently. No `eval`, no dynamic import, no shell-out. |
| T-32-06 | S (Spoofing) | stdin JSON parsing | mitigate | JSON.parse in try/catch; payload shape checked (`typeof payload?.prompt === 'string'`). Malformed input → exit 0 silently. |
</threat_model>
