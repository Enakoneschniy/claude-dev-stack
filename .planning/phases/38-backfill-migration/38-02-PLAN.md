    import { createHash } from 'node:crypto';

    export interface MockResponse {
      observations: Array<{
        type: 'decision' | 'blocker' | 'todo' | 'file_touch' | 'insight';
        content: string;
        entities: Array<{ name: string; type: string }>;
      }>;
      tokens: { input: number; output: number };
      cost_usd: number;
    }

    export function createMockDispatchAgent(fixtures: Record<string, MockResponse> = {}) {
      return async function mockDispatchAgent(opts: { model: string; system?: string; prompt: string; tools?: unknown }) {
        const key = createHash('sha256').update(opts.prompt).digest('hex');
        const fixture = fixtures[key];
        if (!fixture) {
          // Fallback: deterministic synthetic response keyed by first 8 hex chars so tests without explicit fixtures still get SOMETHING.
          return {
            output: JSON.stringify({
              observations: [
                { type: 'decision', content: 'synthetic-decision-' + key.slice(0, 8), entities: [{ name: 'Synthetic', type: 'tool' }] },
              ],
            }),
            tokens: { input: 100, output: 30 },
            cost_usd: 0.000250,
          };
        }
        return {
          output: JSON.stringify({ observations: fixture.observations }),
          tokens: fixture.tokens,
          cost_usd: fixture.cost_usd,
        };
      };
    }

    /** Force an error branch ŌĆö returns a mock that always throws. */
    export function createThrowingMockDispatchAgent(error: Error = new Error('mock dispatch failure')) {
      return async function throwingMockDispatchAgent() {
        throw error;
      };
    }

    /** Counter-wrapped mock ŌĆö exposes `.callCount` for assertion. */
    export function createCountingMockDispatchAgent() {
      let callCount = 0;
      const fn = async (opts: { model: string; prompt: string; system?: string; tools?: unknown }) => {
        callCount++;
        return {
          output: JSON.stringify({ observations: [{ type: 'insight' as const, content: 'call-' + callCount, entities: [] }] }),
          tokens: { input: 50, output: 20 },
          cost_usd: 0.00015,
        };
      };
      (fn as typeof fn & { callCount: () => number }).callCount = () => callCount;
      return fn as typeof fn & { callCount: () => number };
    }

    import { tmpdir } from 'node:os';
    import { join, dirname } from 'node:path';
    import { fileURLToPath } from 'node:url';

    const __dirname = dirname(fileURLToPath(import.meta.url));

    export interface TempVault {
      vaultPath: string;
      projectName: string;
      sessionsDir: string;
      cleanup: () => void;
    }

    export function createTempVault(opts: { projectName?: string; fixtures?: string[] } = {}): TempVault {
      const projectName = opts.projectName ?? 'demo';
      const vaultPath = mkdtempSync(join(tmpdir(), 'cds-migrate-test-'));
      const sessionsDir = join(vaultPath, 'projects', projectName, 'sessions');
      mkdirSync(sessionsDir, { recursive: true });

      if (opts.fixtures) {
        const fixturesRoot = join(__dirname, '..', 'fixtures', 'backfill');
        for (const fixture of opts.fixtures) {
          const src = join(fixturesRoot, fixture);
          const dst = join(sessionsDir, fixture);
          copyFileSync(src, dst);
        }
      }

      return {
        vaultPath,
        projectName,
        sessionsDir,
        cleanup: () => rmSync(vaultPath, { recursive: true, force: true }),
      };
    }

    /** Overwrite a fixture file's content (for hash-changed tests). */

    ```ts
    import { createHash } from 'node:crypto';
    import { readFileSync } from 'node:fs';

    /**
     * Returns the SHA-256 hex digest of a file's raw bytes.
     * Used by Phase 38 backfill to detect if a markdown session has been edited
     * since its last migration (D-95 / D-96).
     *
     * @param path Absolute or relative path to file.
     * @returns 64-character lowercase hex string.
     */
    export function hashFile(path: string): string {
      const buf = readFileSync(path);
      return createHash('sha256').update(buf).digest('hex');
    }

    /** Hash a string directly (used for prompt dedup keys in tests). */
    export function hashString(input: string): string {
      return createHash('sha256').update(input, 'utf8').digest('hex');
    }
    ```

    /**
     * Phase 38 token estimation for Haiku extraction cost preview.
     *
     * Model pricing (Haiku 4.5, per Phase 34 D-28):
     *   Input:  $1.00 per million tokens
     *   Output: $5.00 per million tokens
     *
     * Tokenizer weights are empirical:
     *   Latin + misc:   ~1 token per 4 characters
     *   Cyrillic:       ~1 token per 2.5 characters
     *
     * Estimates err on the HIGH side to avoid sticker shock on actual apply.
     */

    const CYRILLIC_RANGE = /[\u0400-\u04FF]/g;
    const LATIN_CHARS_PER_TOKEN = 4;
    const CYRILLIC_CHARS_PER_TOKEN = 2.5;

    /** Estimate Haiku input-token count for a markdown string. */
    export function estimateTokens(markdown: string): number {
      if (markdown === '') return 0;
      const cyrillic = (markdown.match(CYRILLIC_RANGE) ?? []).length;
      const latinAndOther = markdown.length - cyrillic;
      return Math.ceil(cyrillic / CYRILLIC_CHARS_PER_TOKEN + latinAndOther / LATIN_CHARS_PER_TOKEN);
    }

    /**
     * Estimate total Haiku dispatch cost in USD for one markdown input.
     * Input: passed directly. Output: upper-bound 200 tokens per file (emit_observations payload).
     */
    export function estimateCost(inputTokens: number): number {
      const OUTPUT_TOKENS_UPPER_BOUND = 200;
      const inputCost = inputTokens * 0.000001;        // $1 / 1M tokens
      const outputCost = OUTPUT_TOKENS_UPPER_BOUND * 0.000005;  // $5 / 1M tokens
      return inputCost + outputCost;
    }

    /** Format USD cost as `$X.XXX` with 3-decimal precision (rounded UP). */
    export function formatCost(usd: number): string {
      const cents = Math.ceil(usd * 1000) / 1000;  // round up to nearest 0.001
      return '$' + cents.toFixed(3);
    }

    /** Format bytes as `X B` or `X.X KB` for dry-run table display. */
    export function formatSize(bytes: number): string {
      if (bytes < 1024) return bytes + ' B';
      return (bytes / 1024).toFixed(1) + ' KB';
    }

    import { statSync } from 'node:fs';

    /**
     * Derive the Phase 38 session ID from a markdown filename.
     * Per D-94: `backfill-` + filename without .md extension.
     *
     * Example: `2026-04-09-sync-and-publish.md` ŌåÆ `backfill-2026-04-09-sync-and-publish`

      const base = basename(filename, '.md');
      if (base.length === 0) throw new Error('extractSessionId: filename has no stem: ' + filename);
      return 'backfill-' + base;

    /**
     * Derive the start_time for a session row.
     * Preference order:
     *   1. Leading YYYY-MM-DD prefix in filename (e.g., `2026-04-09-title.md`)
     *   2. File mtime from statSync
     *
     * Returns ISO 8601 string (YYYY-MM-DDTHH:mm:ss.sssZ).
     */
    export function extractStartTime(filePath: string): string {
      const base = basename(filePath, '.md');
      const dateMatch = base.match(/^(\d{4}-\d{2}-\d{2})/);
      if (dateMatch) {
        // Midnight UTC on that date ŌĆö deterministic across timezones.
        return dateMatch[1] + 'T00:00:00.000Z';
      }
      // Fallback to file mtime
      try {
        const stat = statSync(filePath);
        return stat.mtime.toISOString();
      } catch {
        return new Date().toISOString();
      }
    }

    /**

     * Uses the first H1 title (`# Session: ... ŌĆö <title>`) if present, else a generic placeholder.

    export function extractSummary(markdown: string, filename: string): string {

    }
