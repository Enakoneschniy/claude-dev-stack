    ```js
    // lib/install/statusline.mjs — Phase 25 LIMIT-05 SC#4/SC#5
    // Registers hooks/cds-statusline.js as Claude Code's global statusLine.

    import { existsSync, readFileSync, writeFileSync, cpSync, chmodSync } from 'fs';
    import { join } from 'path';
    import { homedir } from 'os';
    import { c, ok, warn, info, step, mkdirp } from '../shared.mjs';

    export function installGlobalStatusline(stepNum, totalSteps, pkgRoot, installState) {
      step(stepNum, totalSteps, 'Installing statusline (cds-statusline.js)');

      const hooksDir = join(homedir(), '.claude', 'hooks');
      mkdirp(hooksDir);

      const src = join(pkgRoot, 'hooks', 'cds-statusline.js');
      const cdsDest = join(hooksDir, 'cds-statusline.js');

      if (!existsSync(src)) {
        warn('Package source hooks/cds-statusline.js missing — skipping statusline install');
        return null;
      }

      cpSync(src, cdsDest);
      try {
        chmodSync(cdsDest, 0o755);
      } catch (err) {
        warn(`Could not set executable bit on ${cdsDest}: ${err.message}`);
      }

      const globalSettingsPath = join(homedir(), '.claude', 'settings.json');
      let settings = {};
      if (existsSync(globalSettingsPath)) {
        try {
          settings = JSON.parse(readFileSync(globalSettingsPath, 'utf8'));
        } catch {
          warn('Global ~/.claude/settings.json is invalid JSON — reinitialising empty object');
          settings = {};
        }
      }

      const newCommand = `node "${cdsDest}"`;
      const existingCommand = settings.statusLine?.command || '';
      const alreadyOurs = existingCommand.includes('cds-statusline.js');

      let previousStatusLine = null;
      if (settings.statusLine && !alreadyOurs) {
        previousStatusLine = JSON.parse(JSON.stringify(settings.statusLine));
        if (installState) installState.previousStatusLine = previousStatusLine;
      }

      settings.statusLine = { type: 'command', command: newCommand };
      writeFileSync(globalSettingsPath, JSON.stringify(settings, null, 2) + '\n');

      const prevLabel = previousStatusLine?.command || '(none)';
      ok(`Statusline: ${c.dim}${prevLabel}${c.reset} → cds-statusline.js (budget enabled)`);

      if (installState) installState.statuslineInstalled = true;
      return previousStatusLine;
    }
    ```
