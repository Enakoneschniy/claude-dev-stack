```javascript
// Public API consumed by CLI and SKILL.md bash block
export async function bridgeSession({
  transcriptPath,     // absolute path to .jsonl OR null (then fallback to most-recent-mtime)
  cwd,                // project cwd — used to derive ~/.claude/projects/{slug}/ dir
  sessionId,          // optional UUID; if set, read {cwd-slug}/{sessionId}.jsonl
  vaultPath,          // absolute vault root
  projectName,        // sanitized project name (no path separators)
  sessionLogPath,     // absolute path to the session log .md just written by SKILL.md
  callHaiku,          // optional injection for tests; defaults to real claude -p invocation
  now = new Date(),   // injection for deterministic tests
}): Promise<{
  newAdrs: Array<{ number: number, topic: string, title: string, path: string }>,
  superseded: Array<{ number: number, topic: string, path: string }>,
  error: string | null,
}>;

// Helper exports (for testability):
export function extractTranscriptText(jsonlPath, maxChars = 600000): string;
export function parseHaikuResponse(rawOutput): { decisions: Array<{topic,title,context,decision,consequences,confidence}> };
export function topicMatchesExistingAdr(topic, decisionsDir): { matched: boolean, filePath?: string, isOldFormat?: boolean };
```

Haiku JSON schema (D-02) — Haiku returns this wrapped in `<decisions>` tags:
```json
{
  "decisions": [
    {
      "topic": "kebab-case-slug",
      "title": "Human readable title",
      "context": "Why it came up",
      "decision": "What was chosen",
      "consequences": "Tradeoffs",
      "confidence": "high|medium|low"
    }
  ]
}
```

<threat_model>

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| session JSONL → Haiku prompt | JSONL content (user/assistant text) is crossed into the Haiku LLM prompt. Adversarial content in the transcript could attempt prompt injection. |
| Haiku output → filesystem | Haiku-returned `topic` and text content are written to files in `vault/projects/{project}/decisions/`. |
| CLI args → subprocess | SKILL.md passes `--session-log`, `--cwd`, `--session-id` into the Node process; Node process spawns `claude` subprocess. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-26-01 | Tampering (Prompt injection) | `buildExtractionPrompt` → Haiku | mitigate | Transcript is treated as data: wrap in a clearly delimited `SESSION TRANSCRIPT:` section at the END of the prompt. Haiku instructions precede the transcript so injection attempts ("ignore previous instructions") appear downstream and are less effective. Additionally: always gate writes on `<decisions>` JSON schema — any output outside the tagged block is discarded by `parseHaikuResponse`. |
| T-26-02 | Tampering (Path traversal via projectName) | `bridgeSession` | mitigate | Reuse existing `bridgeDecisions` guard: reject `projectName` matching `/[/\\]|\.\./`. Throw before any fs access. |
| T-26-03 | Tampering (Path traversal via Haiku-returned topic slug) | `buildAdrBody` filename construction | mitigate | `sanitizeTopicSlug` strips all non-`[a-z0-9-]` chars, collapses dashes, trims leading/trailing dashes. Empty-post-sanitize → decision discarded. Filename is always `{4-digit-number}-{sanitized-topic}.md` joined with `path.join(decisionsDir, filename)`. Test 10 covers `../../../etc/passwd` case. |
| T-26-04 | Tampering (Shell injection via `claude -p` args) | `callHaikuDefault` | mitigate | Use `execFileSync` with an args ARRAY (not `exec` with concatenated string). Model ID and max-tokens are hardcoded constants. Transcript content is passed via `input:` stdin, not as a CLI arg. No interpolated shell metachars reach the subprocess. |
| T-26-05 | DoS (Oversized transcript OOM) | `extractTranscriptText` | mitigate | Hard cap `maxChars = 600_000` (≈150k tokens). Tail-truncate beyond this. Haiku subprocess has `timeout: 60_000`. |
| T-26-06 | Information Disclosure (Session secrets written to vault) | Haiku extraction | accept | The session transcript may contain secrets the user typed or tool outputs. Vault is user-local only, not network-exposed. User already trusts the session log flow (`update-context.mjs` writes the same content). No new disclosure surface introduced by this plan. |
| T-26-07 | Denial of service (Haiku unavailable blocks /end) | `bridgeSession` | mitigate | Fail-open per D-06: catch ALL errors, return `{error}` object, exit 0 from CLI mode. SKILL.md treats non-zero exit as silent failure (`|| true`). /end flow never blocks. |

</threat_model>
