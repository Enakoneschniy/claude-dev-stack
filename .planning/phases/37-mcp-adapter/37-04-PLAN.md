    Exports:

    ```js
    // lib/install/mcp.mjs — Phase 37: register CDS MCP server in each configured project's .claude/settings.json

    import { existsSync, readFileSync, writeFileSync } from 'fs';
    import { join } from 'path';
    import { homedir } from 'os';
    import { c, ok, warn, info } from '../shared.mjs';

    const CDS_SERVER_KEY = 'cds';
    const CDS_SERVER_CONFIG = {
      command: 'claude-dev-stack',
      args: ['mcp', 'serve'],
    };

    export function installCdsMcpServer(stepNum, totalSteps, projectsData) {
      info(`Step ${stepNum}/${totalSteps}: Register CDS MCP server`);
      const projects = (projectsData?.projects || []).filter(p => p.path && existsSync(p.path));

      if (projects.length === 0) {
        warn('No projects to configure — skipping CDS MCP server registration');
        return;
      }

      for (const project of projects) {
        const projectClaudeDir = join(project.path, '.claude');
        const settingsPath = join(projectClaudeDir, 'settings.json');
        const result = _writeMcpEntry(settingsPath);
        if (result === 'added') {
          ok(`CDS MCP server registered for ${c.cyan}${project.name}${c.reset}`);
        } else if (result === 'updated') {
          warn(`Updated existing 'cds' MCP entry in ${project.name} to current recommended configuration`);
        } else if (result === 'skipped-corrupt') {
          warn(`settings.json corrupt in ${project.name} — skipped MCP registration`);
        } else if (result === 'no-op') {
          info(`CDS MCP server already registered for ${project.name} — no change`);
        }
      }
    }

    // Exported for unit tests
    export function _writeMcpEntry(settingsPath) {
      if (!existsSync(settingsPath)) {
        // Create minimal settings with just the mcp.servers entry
        const newSettings = { 'mcp.servers': { [CDS_SERVER_KEY]: { ...CDS_SERVER_CONFIG } } };
        writeFileSync(settingsPath, JSON.stringify(newSettings, null, 2) + '\n');
        return 'added';
      }

      let settings;
      try {
        settings = JSON.parse(readFileSync(settingsPath, 'utf8'));
      } catch {
        return 'skipped-corrupt';
      }

      // Use the flat "mcp.servers" key per Claude Code format verified in RESEARCH §4.1
      if (!settings['mcp.servers']) settings['mcp.servers'] = {};
      const existing = settings['mcp.servers'][CDS_SERVER_KEY];
      if (existing) {
        if (existing.command === CDS_SERVER_CONFIG.command &&
            Array.isArray(existing.args) &&
            existing.args.length === CDS_SERVER_CONFIG.args.length &&
            existing.args.every((a, i) => a === CDS_SERVER_CONFIG.args[i])) {
          return 'no-op';
        }
        settings['mcp.servers'][CDS_SERVER_KEY] = { ...CDS_SERVER_CONFIG };
        writeFileSync(settingsPath, JSON.stringify(settings, null, 2) + '\n');
        return 'updated';
      }

      settings['mcp.servers'][CDS_SERVER_KEY] = { ...CDS_SERVER_CONFIG };
      writeFileSync(settingsPath, JSON.stringify(settings, null, 2) + '\n');
      return 'added';
    }
    ```

    Critical rules:
    - Use the flat key `'mcp.servers'` (string literal containing a dot) per RESEARCH §4.1.
    - Never touch other `mcp.servers.*` entries besides `cds`.
    - Never touch non-`mcp.servers` keys (hooks, permissions, etc.).
    - On corrupt JSON, log warning and SKIP the file — NEVER clobber.
    - Always trailing newline on write (consistency with hooks.mjs pattern).
  </action>
