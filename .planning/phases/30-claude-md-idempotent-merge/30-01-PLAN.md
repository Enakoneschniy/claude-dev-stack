    ```javascript
    /**
     * Update {projectPath}/CLAUDE.md idempotently with arbitrary managed content.
     * Wraps `managedContent` in MARKER_START / MARKER_END.
     *
     * - If CLAUDE.md doesn't exist → create it with our section
     * - If markers exist → replace content between them
     * - If markers don't exist → append our section at the end
     *
     * Threat model:
     *   T-30-01 (accepted) — if user-authored content happens to contain literal
     *   `<!-- @claude-dev-stack:start -->` / `<!-- @claude-dev-stack:end -->`
     *   lines, this function treats that span as managed and replaces it.
     *   Markdown-aware parsing is YAGNI (CONTEXT.md deferred).
     *
     * @param {string} projectPath - absolute path to project directory
     * @param {string} managedContent - section body (without markers; markers added here)
     * @returns {'created' | 'updated' | 'appended' | 'unchanged'}
     */
    export function updateManagedSection(projectPath, managedContent) {
      const claudeMdPath = join(projectPath, 'CLAUDE.md');
      const newSection = `${MARKER_START}\n${managedContent.trim()}\n${MARKER_END}`;

      if (!existsSync(claudeMdPath)) {
        const content = `# CLAUDE.md\n\nThis file guides Claude Code when working in this project.\n\n${newSection}\n`;
        writeFileSync(claudeMdPath, content);
        return 'created';
      }

      const existing = readFileSync(claudeMdPath, 'utf8');

      if (existing.includes(MARKER_START) && existing.includes(MARKER_END)) {
        const escaped = (s) => s.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
        const pattern = new RegExp(`${escaped(MARKER_START)}[\\s\\S]*?${escaped(MARKER_END)}`, 'g');
        const updated = existing.replace(pattern, newSection);
        if (updated === existing) return 'unchanged';
        writeFileSync(claudeMdPath, updated);
        return 'updated';
      }

      const separator = existing.endsWith('\n') ? '\n' : '\n\n';
      const updated = existing + separator + newSection + '\n';
      writeFileSync(claudeMdPath, updated);
      return 'appended';
    }
    ```
