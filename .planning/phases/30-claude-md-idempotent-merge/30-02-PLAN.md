    ```javascript
    /**
     * Format the per-project CLAUDE.md status line.
     * Pure function ŌĆö returns the string without writing to console.
     *
     * @param {string} projectName
     * @param {'created' | 'updated' | 'appended' | 'unchanged' | string} status
     * @returns {string}
     */
    export function formatClaudeMdStatus(projectName, status) {
      switch (status) {
        case 'created':   return `${projectName} ŌåÆ CLAUDE.md: created`;
        case 'updated':   return `${projectName} ŌåÆ CLAUDE.md: updated`;
        case 'appended':  return `${projectName} ŌåÆ CLAUDE.md: appended (existing content preserved)`;
        case 'unchanged': return `${projectName} ŌåÆ CLAUDE.md: unchanged`;
        default:          return `${projectName} ŌåÆ CLAUDE.md: ${status}`; // defensive ŌĆö never includes 'overwritten'
      }
    }

    /**
     * Print the status using the appropriate color helper from lib/shared.mjs.
     */
    export function printClaudeMdStatus(projectName, status) {
      const line = formatClaudeMdStatus(projectName, status);
      switch (status) {
        case 'created':   info(line); break;
        case 'updated':   ok(line);   break;
        case 'appended':  warn(line); break;
        case 'unchanged': info(line); break;
        default:          info(line); break;
      }
    }
    ```
